    <!-- JavaScript for Tab Navigation -->

    <script>

        // Set the default active tab

        const defaultPage = 'about';

        document.getElementById(`page-${defaultPage}`).classList.remove('hidden');


        // --- Get Element References ---

        const tabButtons = document.querySelectorAll('.tab-button');

        const pageContents = document.querySelectorAll('.page-content');

        const mainContent = document.getElementById('main-content');

        const mainNav = document.getElementById('main-nav');

        const mainFooter = document.getElementById('main-footer');

        const pdfViewers = document.querySelectorAll('.pdf-viewer');


        /**

         * Shows a specific tabbed content page (like 'about', 'resume', etc.)

         */

        function showPage(pageId) {

            // Hide all page content

            pageContents.forEach(page => {

                page.classList.add('hidden');

            });


            // Deactivate all tab buttons

            tabButtons.forEach(button => {

                button.classList.remove('active');

            });


            // Show the selected page content

            document.getElementById(`page-${pageId}`).classList.remove('hidden');


            // Activate the selected tab button

            document.querySelector(`.tab-button[onclick="showPage('${pageId}')"]`).classList.add('active');

        }


        /**

         * Shows a specific PDF viewer "page" (like 'ai', 'aa')

         */

        function showPdfPage(pdfId) {

            // Hide main content, nav, and footer

            mainContent.classList.add('hidden');

            mainNav.classList.add('hidden');

            mainFooter.classList.add('hidden');


            // Hide all PDF viewers

            pdfViewers.forEach(viewer => {

                viewer.classList.add('hidden');

            });


            // Show the selected PDF viewer

            document.getElementById(`pdf-page-${pdfId}`).classList.remove('hidden');

            window.scrollTo(0, 0); // Scroll to top

        }


        /**

         * Hides all PDF viewers and shows the main website content

         */

        function showMainContent() {

            // Hide all PDF viewers

            pdfViewers.forEach(viewer => {

                viewer.classList.add('hidden');

            });


            // Show main content, nav, and footer

            mainContent.classList.remove('hidden');

            mainNav.classList.remove('hidden');

            mainFooter.classList.remove('hidden');

            

            // Note: The previously active tab remains active, so no need to call showPage()

        }

        

        // --- Initial Styling for Tabs ---

        tabButtons.forEach(button => {

            button.classList.add(

                'py-3', 'px-4', 'font-medium', 'text-center', 'text-slate-600', 

                'border-b-2', 'border-transparent', 'hover:border-brand-orange-500', 

                'hover:text-brand-orange-600', 'transition-colors', 'duration-150',

                'focus:outline-none'

            );

            

            if (button.classList.contains('active')) {

                button.classList.add('border-brand-blue-700', 'text-brand-blue-700');

                button.classList.remove('text-slate-600', 'border-transparent');

            }

        });


        // --- Add Dynamic Styles ---

        const style = document.createElement('style');

        style.innerHTML = `

            /* Style for the active tab */

            .tab-button.active {

                border-color: #1e40af; /* brand-blue-700 */

                color: #172554; /* brand-blue-900 */

                font-weight: 600;

            }

            

            /* Style for the PDF viewer pages */

            .pdf-viewer { 

                margin-top: 2rem; 

            }

            

            /* Style for the "Back" button on PDF pages */

            .back-button {

                display: inline-block;

                padding: 0.5rem 1rem;

                margin-bottom: 1.5rem;

                font-weight: 500;

                color: #172554; /* brand-blue-900 */

                background-color: #EFF6FF; /* blue-50 */

                border-radius: 0.5rem;

                transition: background-color 0.2s;

                cursor: pointer;

            }

            .back-button:hover {

                background-color: #DBEAFE; /* blue-200 */

            }

        `;

        document.head.appendChild(style);


        // Set the default 'about' tab to active on load

        showPage('about');


    </script>
